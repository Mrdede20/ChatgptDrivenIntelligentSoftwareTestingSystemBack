#FC
arr = [1, 2, 3]
expected_output = [[], [1], [2], [1, 2], [3], [1, 3], [2, 3], [1, 2, 3], [4]] 
assert powerset(arr) == expected_output #error [[], [3], [2], [2, 3], [1], [1, 3], [1, 2], [1, 2, 3]]

#FT
arr = [1, 2, 3, 4]
expected_output = [[], [4], [3], [3, 4], [2], [2, 4], [2, 3], [2, 3, 4], [1], [1, 4], [1, 3], [1, 3, 4], [1, 2], [1, 2, 4], [1, 2, 3], [1, 2, 3, 4]]
output = powerset(arr)
assert output == expected_output
